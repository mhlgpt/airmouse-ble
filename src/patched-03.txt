#include <Arduino.h>
#include <Wire.h>
#include <BleMouse.h>

#define LED_PIN 2

BleMouse bleMouse("ESP32 Air Mouse", "Espressif", 100);

// --- RAW I2C SETTINGS ---
const int MPU_ADDR = 0x68; // Change to 0x69 if the blue light flashes!

// Raw sensor values are much larger numbers, so we need higher tuning values
const int DEADZONE = 300;     // Prevents drift. Increase if mouse moves on its own
const int SENSITIVITY = 400;  // Decrease to make mouse faster, increase to make it slower

void setup() {
  Serial.begin(115200);
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, HIGH); // LED solid = booting
  
  // Anti-Noise Fix
  pinMode(21, INPUT_PULLUP);
  pinMode(22, INPUT_PULLUP);
  Wire.begin(21, 22); 
  Wire.setClock(10000); 
  delay(100); 

  Serial.println("\n--- BOOTING RAW I2C MODE ---");

  // Wake up the MPU chip directly by writing a 0 to its Power Management Register
  Wire.beginTransmission(MPU_ADDR);
  Wire.write(0x6B); // The Power Management Register
  Wire.write(0x00); // Send 0 to wake it up
  byte error = Wire.endTransmission();

  if (error == 0) {
      Serial.println("Sensor successfully woken up! Bypassed library checks.");
  } else {
      Serial.println("CRITICAL ERROR: Cannot reach sensor. Check wires or change MPU_ADDR to 0x69.");
      while (1) {
          digitalWrite(LED_PIN, !digitalRead(LED_PIN)); // Fast blink = error
          delay(100);
      }
  }

  Serial.println("Starting Bluetooth...");
  bleMouse.begin();
  
  digitalWrite(LED_PIN, LOW); // LED off = ready to fly!
  Serial.println("--- READY! WAITING FOR BLUETOOTH CONNECTION ---");
}

void loop() {
  if (bleMouse.isConnected()) {
    
    // Ask the chip specifically for the 6 bytes of Gyroscope data
    Wire.beginTransmission(MPU_ADDR);
    Wire.write(0x43); // Starting register for Gyroscope data
    Wire.endTransmission(false);
    Wire.requestFrom(MPU_ADDR, 6, true); 

    // If we received the 6 bytes, do the math
    if (Wire.available() == 6) {
      // Combine the raw High and Low bytes into single numbers
      int16_t gyroX = (Wire.read() << 8 | Wire.read());
      int16_t gyroY = (Wire.read() << 8 | Wire.read());
      int16_t gyroZ = (Wire.read() << 8 | Wire.read());
      
      // Calculate mouse movement
      int moveX = -gyroZ / SENSITIVITY; 
      int moveY = -gyroY / SENSITIVITY;
      
      // Apply deadzone
      if (abs(gyroZ) < DEADZONE) moveX = 0;
      if (abs(gyroY) < DEADZONE) moveY = 0;
      
      // Move the mouse!
      if (moveX != 0 || moveY != 0) {
        bleMouse.move(moveX, moveY);
      }
    }
  }
  delay(20); // 50Hz polling rate
}
