#include <Arduino.h>
#include <Wire.h>
#include <BleMouse.h>
#include "MPU9250.h"

#define LED_PIN 2

BleMouse bleMouse("ESP32 Air Mouse", "Espressif", 100);
MPU9250 mpu;

// --- Tuning Variables ---
const int DEADZONE = 10;     // Increase if the mouse jitters when your hand is still
const int SENSITIVITY = 10;  // Decrease to make the mouse faster

void setup() {
  Serial.begin(115200);
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, HIGH); // LED solid = booting
  
  // --- THE MAGIC NOISE FIX ---
  // Turn on internal pull-up resistors to stabilize the wires
  pinMode(21, INPUT_PULLUP);
  pinMode(22, INPUT_PULLUP);
  
  Wire.begin(21, 22); 
  Wire.setClock(10000); // Slow down the I2C clock to cut through the static!
  delay(1000); 
  // ---------------------------

  Serial.println("\n--- SYSTEM BOOTING ---");

  // Auto-Detect Address (just to be safe)
  Serial.println("Checking MPU9250...");
  if (!mpu.setup(0x68)) {  
      if (!mpu.setup(0x69)) {
          Serial.println("CRITICAL ERROR: MPU9250 not found.");
          while (1) {
              digitalWrite(LED_PIN, !digitalRead(LED_PIN)); // Fast blink = error
              delay(100);
          }
      }
  }

  Serial.println("Sensor found! Calibrating... Keep it perfectly flat and still!");
  delay(2000);
  mpu.calibrateAccelGyro();
  
  Serial.println("Calibration complete. Starting Bluetooth...");
  bleMouse.begin();
  
  digitalWrite(LED_PIN, LOW); // LED off = ready to fly!
  Serial.println("--- READY! WAITING FOR BLUETOOTH CONNECTION ---");
}

void loop() {
  // Only process movement if a device is paired and connected
  if (bleMouse.isConnected()) {
    
    if (mpu.update()) {
      float gyroY = mpu.getGyroY(); // Roll (left/right tilt)
      float gyroZ = mpu.getGyroZ(); // Yaw (left/right spin)
      
      int moveX = -(int)gyroZ / SENSITIVITY; 
      int moveY = -(int)gyroY / SENSITIVITY;
      
      if (abs(gyroZ) < DEADZONE) moveX = 0;
      if (abs(gyroY) < DEADZONE) moveY = 0;
      
      if (moveX != 0 || moveY != 0) {
        bleMouse.move(moveX, moveY);
      }
    }
  }
  delay(20); // Run at 50Hz
}
